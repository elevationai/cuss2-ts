Base Component Class

All components inherit from the base Component class, providing common functionality:

// Common component methods
component.enable();     // Enable the component
component.disable();    // Disable the component
component.query();      // Query component status
component.cancel();     // Cancel current operation

// Component properties
component.id           // Unique component identifier
component.meta         // Component metadata
component.status       // Current status (READY, UNAVAILABLE, ERROR, etc.)

// Event handling
component.on('statusChange', (status) => {
    console.log(`Component status: ${status}`);
});

component.on('error', (error) => {
    console.error(`Component error: ${error.message}`);
});

üñ®Ô∏è Printer Components

Printers support various output formats including raw data, images, and structured documents.

BoardingPassPrinter

const bpp = cuss2.boardingPassPrinter;

// Print raw data
await bpp.printRaw(base64Data);

// Print boarding pass template
await bpp.printBoardingPass({
    passengerName: "SMITH/JOHN",
    flight: "AA123",
    seat: "12A",
    gate: "B42",
    boardingTime: "10:30"
});

// Printer with feeder control
if (bpp.feeder) {
    await bpp.feeder.setup({ mediaSize: "BP_2x6" });
    await bpp.feeder.takeMedia();
}

// Handle print events
bpp.on('printComplete', () => {
    console.log('Boarding pass printed successfully');
});

bpp.on('paperOut', () => {
    console.error('Printer out of paper');
});

BagTagPrinter

const btp = cuss2.bagTagPrinter;

// Print bag tag with airline data
await btp.printRaw(iataFormattedData);

// Query printer status
const status = await btp.query();
if (status === ComponentStatus.READY) {
    await btp.printBagTag({
        tagNumber: "AA123456789",
        destination: "LAX",
        routing: ["DFW", "LAX"],
        passenger: "SMITH/J"
    });
}

// Dispenser control for tag output
if (btp.dispenser) {
    await btp.dispenser.present();  // Present tag to passenger
    await btp.dispenser.retract();  // Retract if not taken
}

üìä Data Reader Components

Components that read various types of data from documents, cards, and biometric sources.

BarcodeReader

const barcode = cuss2.barcodeReader;

// Enable and start reading
await barcode.enable();

// Handle barcode data
barcode.on('data', (scanData) => {
    console.log('Barcode scanned:', scanData.value);
    console.log('Barcode type:', scanData.symbology);
    
    // Process based on type
    if (scanData.symbology === 'PDF417') {
        processBoardingPass(scanData.value);
    }
});

// Multiple barcode types
barcode.on('dataList', (barcodes) => {
    barcodes.forEach(scan => {
        console.log(`Found ${scan.symbology}: ${scan.value}`);
    });
});

PassportReader

const passport = cuss2.passportReader;

// Enable with specific settings
await passport.enable();

// Read passport data
passport.on('data', (passportData) => {
    console.log('Document Number:', passportData.documentNumber);
    console.log('Name:', passportData.surname, passportData.givenNames);
    console.log('Nationality:', passportData.nationality);
    console.log('Expiry:', passportData.expiryDate);
    
    // Verify MRZ checksum
    if (passportData.mrzValid) {
        console.log('Document verified');
    }
});

// Handle chip reading
passport.on('chipData', (chipData) => {
    console.log('Biometric data available');
    // Process facial image, fingerprints, etc.
});

üí≥ Payment Components

Handle various payment methods including cards, cash, and coins.

PaymentDevice

const payment = cuss2.paymentDevice;

// Initialize payment for amount
await payment.startTransaction({
    amount: 125.50,
    currency: "USD",
    type: "PURCHASE"
});

// Handle payment events
payment.on('cardInserted', () => {
    console.log('Processing card...');
});

payment.on('approved', (transaction) => {
    console.log('Payment approved:', transaction.authCode);
    await payment.printReceipt();
});

payment.on('declined', (reason) => {
    console.error('Payment declined:', reason);
});

// Cancel transaction if needed
await payment.cancel();

CashAcceptor

const cash = cuss2.cashAcceptor;

// Enable cash acceptance
await cash.enable();
await cash.startAccepting(100.00); // Accept up to $100

// Track inserted bills
cash.on('billInserted', (bill) => {
    console.log(`Inserted: $${bill.value}`);
    totalInserted += bill.value;
});

// Complete transaction
cash.on('complete', (total) => {
    console.log(`Total cash received: $${total}`);
});

Component Status Management

All components report their status and can be monitored for availability:

// Monitor all components
cuss2.on('componentStatusChange', (event) => {
    console.log(`${event.componentId}: ${event.status}`);
    
    // Update UI based on status
    switch(event.status) {
        case ComponentStatus.READY:
            enableFeature(event.componentId);
            break;
        case ComponentStatus.UNAVAILABLE:
            disableFeature(event.componentId);
            break;
        case ComponentStatus.ERROR:
            showErrorMessage(event.componentId, event.error);
            break;
    }
});

// Check specific component availability
if (cuss2.boardingPassPrinter?.status === ComponentStatus.READY) {
    showPrintOption();
}

// Wait for component to be ready
async function waitForComponent(component) {
    return new Promise((resolve) => {
        if (component.status === ComponentStatus.READY) {
            resolve();
        } else {
            component.once('statusChange', (status) => {
                if (status === ComponentStatus.READY) {
                    resolve();
                }
            });
        }
    });
}

Events and Error Handling

Comprehensive event system for monitoring component state and handling errors:

// Component-specific events
component.on('statusChange', handler);
component.on('data', handler);
component.on('error', handler);

// Global component events
cuss2.on('componentEnabled', ({ componentId }) => {
    console.log(`${componentId} enabled`);
});

cuss2.on('componentDisabled', ({ componentId }) => {
    console.log(`${componentId} disabled`);
});

// Error handling patterns
try {
    await component.enable();
} catch (error) {
    if (error.code === 'DEVICE_OFFLINE') {
        // Handle offline device
        showMaintenanceMessage();
    } else if (error.code === 'PERMISSION_DENIED') {
        // Handle permission issues
        requestPermission();
    } else {
        // Generic error handling
        console.error('Component error:', error);
    }
}

// Retry logic for transient failures
async function enableWithRetry(component, maxRetries = 3) {
    for (let i = 0; i < maxRetries; i++) {
        try {
            await component.enable();
            return;
        } catch (error) {
            console.warn(`Attempt ${i + 1} failed:`, error);
            if (i === maxRetries - 1) throw error;
            await new Promise(r => setTimeout(r, 1000 * Math.pow(2, i)));
        }
    }
}
